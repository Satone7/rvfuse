# C920 ONNX Runtime Cross-Compilation Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Create `tools/c920-onnxrt/` with a Dockerfile and build.sh that cross-compiles a full (non-minimal) ONNX Runtime v1.17.3 for RISC-V using the pre-built LLVM 13 toolchain.

**Architecture:** An x86_64 Ubuntu 22.04 Docker container mounts the LLVM 13 toolchain, ORT source, and Eigen. It uses a CMake toolchain file targeting `riscv64-unknown-linux-gnu` with a sysroot extracted from `riscv64/ubuntu:22.04`. The build script orchestrates sysroot extraction, image build, cross-compilation, YOLO runner linking, and artifact export.

**Tech Stack:** Docker, CMake, Ninja, Clang 13 (LLVM), ONNX Runtime v1.17.3, Eigen 3.4.0, Bash

### Task 1: Create toolchain file

**Files:**
- Create: `tools/c920-onnxrt/riscv64-linux-clang13.cmake`

**Context:** ONNX Runtime ships `cmake/linux_arm64_crosscompile_toolchain.cmake` as a reference. We adapt it for RISC-V + Clang 13. This file tells CMake how to cross-compile: compiler paths, target triple, sysroot search paths.

**Step 1: Create the toolchain file**

```cmake
# Cross-compilation toolchain for RISC-V 64-bit using Clang 13
SET(CMAKE_SYSTEM_NAME Linux)
SET(CMAKE_SYSTEM_VERSION 1)
SET(CMAKE_SYSTEM_PROCESSOR riscv64)

SET(CMAKE_C_COMPILER $ENV{LLVM_INSTALL}/bin/clang-13)
SET(CMAKE_CXX_COMPILER $ENV{LLVM_INSTALL}/bin/clang++-13)

SET(CMAKE_C_COMPILER_TARGET riscv64-unknown-linux-gnu)
SET(CMAKE_CXX_COMPILER_TARGET riscv64-unknown-linux-gnu)

SET(CMAKE_FIND_ROOT_PATH $ENV{SYSROOT})

SET(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
SET(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
SET(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
SET(CMAKE_FIND_ROOT_PATH_MODE_PACKAGE ONLY)

# Use ld.lld for linking (avoids missing GCC cross-linker)
SET(CMAKE_EXE_LINKER_FLAGS "-fuse-ld=lld-13")
SET(CMAKE_SHARED_LINKER_FLAGS "-fuse-ld=lld-13")
```

Key notes:
- Uses environment variables (`LLVM_INSTALL`, `SYSROOT`) so the build script can inject paths at build time
- `FIND_ROOT_PATH_MODE_PROGRAM NEVER` allows finding host tools (cmake, python, protobuf-compiler) while restricting library/include search to sysroot
- `fuse-ld=lld-13` avoids needing a GCC cross-linker installed

```bash

git commit -m "feat(c920-onnxrt): add RISC-V cross-compilation CMake toolchain file"
```

### Task 2: Create Dockerfile

**Files:**
- Create: `tools/c920-onnxrt/Dockerfile`

**Context:** The Dockerfile builds on `ubuntu:22.04` and installs all host-side build dependencies needed for a full ONNX Runtime build. It does NOT use `--platform riscv64` — the container runs on x86_64 and cross-compiles via Clang 13.

**Step 1: Create the Dockerfile**

```dockerfile
FROM ubuntu:22.04

ARG DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y --no-install-recommends \
        cmake \
        ninja-build \
        python3 \
        python3-pip \
        git \
        ca-certificates \
        make \
        libprotobuf-dev \
        protobuf-compiler \
        libflatbuffers-dev \
        && rm -rf /var/lib/apt/lists/

# Work directories
WORKDIR /build

# ORT source, Eigen, LLVM toolchain, sysroot, and YOLO runner are
# mounted via volumes at runtime (see build.sh).
```

Notes:
- `libprotobuf-dev` + `protobuf-compiler`: ONNX Runtime full build uses Protobuf
- `libflatbuffers-dev`: ORT uses FlatBuffers for model serialization
- No `RUN cmake` here — cmake runs inside the build script's `docker run` so we can pass volume paths
- We intentionally avoid installing `clang-13` from apt — we mount the pre-built `llvm-install/` instead

### Task 3: Create build.sh — prerequisites and argument parsing

**Files:**

**Context:** This is the main entry point. Follow the patterns from `tools/rv64gcv-onnxrt/build.sh` (SCRIPT_DIR, PROJECT_ROOT, set -euo pipefail). The script will be implemented incrementally across Tasks 3-6.

**Step 1: Create build.sh with header, argument parsing, and prerequisite checks**

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "${SCRIPT_DIR}/../.." && pwd)"
OUTPUT_DIR="${PROJECT_ROOT}/output/c920-ort"
VENDOR_DIR="${PROJECT_ROOT}/tools/rv64gcv-onnxrt/vendor"
LLVM_INSTALL="${PROJECT_ROOT}/third_party/llvm-install"
ORT_SOURCE="${VENDOR_DIR}/onnxruntime"
EIGEN_SOURCE="${VENDOR_DIR}/eigen"
YOLO_RUNNER="${PROJECT_ROOT}/tools/yolo_runner"
DOCKER_IMAGE="rvfuse/c920-onnxrt-builder:latest"

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

info()  { echo -e "${GREEN}=== $* ===${NC}"; }
warn()  { echo -e "${YELLOW}Warning: $*${NC}"; }
error() { echo -e "${RED}Error: $*${NC}" >&2; exit 1; }

# --- Argument parsing ---
FORCE=false
SKIP_SYSROOT=false
JOBS=$(nproc 2>/dev/null || echo 4)

while [[ $# -gt 0 ]]; do
    case "$1" in
        --force)         FORCE=true; shift ;;
        --skip-sysroot)  SKIP_SYSROOT=true; shift ;;
        -j|--jobs)       JOBS="$2"; shift 2 ;;
        -j*)             JOBS="${1#-j}"; shift ;;
        *)               error "Unknown argument: $1" ;;
    esac
done

# --- Step 0: Prerequisites ---
check_prerequisites() {
    info "Checking prerequisites..."
    command -v docker &>/dev/null || error "Docker not found. Please install Docker."
    docker info &>/dev/null || error "Docker daemon not running."

    [ -d "${LLVM_INSTALL}/bin" ] || error "LLVM install not found at ${LLVM_INSTALL}"
    [ -d "${ORT_SOURCE}/cmake" ] || error "ORT source not found at ${ORT_SOURCE}. Run tools/rv64gcv-onnxrt/build.sh first or clone manually."
    [ -d "${EIGEN_SOURCE}" ]     || error "Eigen not found at ${EIGEN_SOURCE}"
    [ -f "${YOLO_RUNNER}/yolo_runner.cpp" ] || error "YOLO runner not found at ${YOLO_RUNNER}"

    info "All prerequisites met."
}

check_prerequisites
```

Notes:
- `--force`: re-extract sysroot, rebuild image, rebuild ORT
- `--skip-sysroot`: skip sysroot extraction (use existing)
- `-j/--jobs`: parallelism control, defaults to nproc
- Follows `SCRIPT_DIR` + `PROJECT_ROOT` pattern from existing build.sh

### Task 4: Add sysroot extraction to build.sh

**Files:**
- Modify: `tools/c920-onnxrt/build.sh` (append after `check_prerequisites`)

**Context:** We need a RISC-V sysroot for the cross-compiler to find headers and libraries. The approach: start a `riscv64/ubuntu:22.04` container, install build dependencies into it, then extract the relevant directories (`/lib`, `/usr/lib`, `/usr/include`).

**Step 1: Append sysroot extraction function**

Add to `build.sh` after the `check_prerequisites` call:

```bash
# --- Step 1: Extract riscv64 sysroot ---
extract_sysroot() {
    local sysroot="${OUTPUT_DIR}/sysroot"

    if [[ "${SKIP_SYSROOT}" == "true" && -d "${sysroot}/usr" ]]; then
        info "Skipping sysroot extraction (--skip-sysroot)"
        return 0
    fi

    if [[ "${FORCE}" != "true" && -d "${sysroot}/usr" ]]; then
        info "Sysroot already exists at ${sysroot}. Use --force to re-extract."
        return 0
    fi

    info "Extracting riscv64 sysroot from riscv64/ubuntu:22.04..."
    rm -rf "${sysroot}"
    mkdir -p "${sysroot}"

    # Create a temporary container to install deps, then export its filesystem
    local tmp_container="rvfuse-sysroot-prep-$$"

    docker run --name "${tmp_container}" -d riscv64/ubuntu:22.04 tail -f /dev/null > /dev/null
    trap "docker rm -f ${tmp_container} 2>/dev/null || true" RETURN

    docker exec "${tmp_container}" apt-get update -qq
    docker exec "${tmp_container}" apt-get install -y --no-install-recommends -qq \
        libc6-dev \
        libstdc++-dev \
        libgcc-s-dev \
        libprotobuf-dev \
        libabsl-dev \
        > /dev/null

    info "Copying sysroot directories from container..."

    # Extract /lib, /usr/lib, /usr/include
    docker cp "${tmp_container}:/lib"    "${sysroot}/lib"
    docker cp "${tmp_container}:/usr/lib"    "${sysroot}/usr_lib_tmp"
    docker cp "${tmp_container}:/usr/include" "${sysroot}/usr_include_tmp"

    docker rm -f "${tmp_container}" > /dev/null
    trap - RETURN

    # Flatten into sysroot layout: /usr/lib and /usr/include
    mkdir -p "${sysroot}/usr"
    mv "${sysroot}/usr_lib_tmp"    "${sysroot}/usr/lib"
    mv "${sysroot}/usr_include_tmp" "${sysroot}/usr/include"

    info "Sysroot extracted to ${sysroot}"
    echo "  $(du -sh "${sysroot}" | cut -f1)"
}
```

Key notes:
- Installs dev packages (libc6-dev, libstdc++-dev, etc.) so the sysroot has both runtime libs and headers
- Uses `docker run -d` + `docker exec` (not `docker commit`) for cleanliness
- Trap ensures container cleanup on error
- Renames `usr_lib_tmp` → `usr/lib` to avoid path conflict with intermediate `docker cp`

### Task 5: Add Docker image build and cross-compilation to build.sh

**Files:**

**Context:** Build the Docker image from the Dockerfile, then run the container with all volumes mounted to perform the actual cross-compilation.

**Step 1: Append Docker image build**

```bash
# --- Step 2: Build cross-compilation Docker image ---
build_image() {
    info "Building cross-compilation Docker image..."
    DOCKER_BUILDKIT=1 docker build \
        -t "${DOCKER_IMAGE}" \
        -f "${SCRIPT_DIR}/Dockerfile" \
        "${SCRIPT_DIR}"
    info "Docker image built: ${DOCKER_IMAGE}"
}
```

**Step 2: Append cross-compilation step**

```bash
# --- Step 3: Cross-compile ONNX Runtime ---
cross_compile() {
    local ort_build="${OUTPUT_DIR}/.build"
    local ort_install="${OUTPUT_DIR}/onnxruntime"

    if [[ "${FORCE}" != "true" && -d "${ort_install}/lib" ]]; then
        info "ORT already built at ${ort_install}. Use --force to rebuild."
        return 0
    fi

    info "Cross-compiling ONNX Runtime (full build)..."
    mkdir -p "${ort_build}" "${ort_install}"

    docker run --rm \
        -v "${LLVM_INSTALL}:/llvm-install:ro" \
        -v "${ORT_SOURCE}:/onnxruntime:ro" \
        -v "${EIGEN_SOURCE}:/eigen:ro" \
        -v "${OUTPUT_DIR}/sysroot:/sysroot:ro" \
        -v "${ort_build}:/build" \
        -v "${ort_install}:/install" \
        -e LLVM_INSTALL=/llvm-install \
        -e SYSROOT=/sysroot \
        -w /build \
        "${DOCKER_IMAGE}" \
        bash -c "
            cmake /onnxruntime/cmake \
                -DCMAKE_TOOLCHAIN_FILE=/onnxruntime/tools/c920-onnxrt/riscv64-linux-clang13.cmake \
                -DCMAKE_INSTALL_PREFIX=/install \
                -DCMAKE_BUILD_TYPE=Release \
                -Donnxruntime_BUILD_SHARED_LIB=ON \
                -Donnxruntime_BUILD_UNIT_TESTS=OFF \
                -Donnxruntime_DISABLE_RTTI=ON \
                -DFETCHCONTENT_SOURCE_DIR_EIGEN=/eigen \
                -DCMAKE_CXX_FLAGS='-Wno-stringop-overflow' \
                -G Ninja \
            && ninja -j${JOBS} \
            && ninja install/strip
        "

    info "ONNX Runtime cross-compiled to ${ort_install}"
}
```

Wait — there's an issue. The `CMAKE_TOOLCHAIN_FILE` path inside the container refers to `/onnxruntime/tools/c920-onnxrt/...` but the toolchain file is at `tools/c920-onnxrt/` in the PROJECT ROOT, not inside the ORT source. We need to mount the project root or the toolchain file separately.

**Corrected approach:** Mount the toolchain file explicitly:

```bash
    docker run --rm \
        -v "${LLVM_INSTALL}:/llvm-install:ro" \
        -v "${ORT_SOURCE}:/onnxruntime:ro" \
        -v "${EIGEN_SOURCE}:/eigen:ro" \
        -v "${OUTPUT_DIR}/sysroot:/sysroot:ro" \
        -v "${ort_build}:/build" \
        -v "${ort_install}:/install" \
        -v "${SCRIPT_DIR}/riscv64-linux-clang13.cmake:/toolchain.cmake:ro" \
        -e LLVM_INSTALL=/llvm-install \
        -e SYSROOT=/sysroot \
        -w /build \
        "${DOCKER_IMAGE}" \
        bash -c "
            cmake /onnxruntime/cmake \
                -DCMAKE_TOOLCHAIN_FILE=/toolchain.cmake \
                -DCMAKE_INSTALL_PREFIX=/install \
                -DCMAKE_BUILD_TYPE=Release \
                -Donnxruntime_BUILD_SHARED_LIB=ON \
                -Donnxruntime_BUILD_UNIT_TESTS=OFF \
                -Donnxruntime_DISABLE_RTTI=ON \
                -DFETCHCONTENT_SOURCE_DIR_EIGEN=/eigen \
                -DCMAKE_CXX_FLAGS='-Wno-stringop-overflow' \
                -G Ninja \
            && ninja -j${JOBS} \
            && ninja install/strip
        "
```

Also note: ORT's `build.sh` (in-repo) is NOT used. We invoke cmake directly on `/onnxruntime/cmake/` which is ORT's standard way of building without their wrapper script.

**Step 3: Append YOLO runner cross-compilation**

```bash
# --- Step 4: Build YOLO runner ---
build_yolo_runner() {
    local ort_install="${OUTPUT_DIR}/onnxruntime"
    local runner_out="${OUTPUT_DIR}/yolo_inference"

    if [[ "${FORCE}" != "true" && -f "${runner_out}" ]]; then
        info "YOLO runner already built. Use --force to rebuild."
        return 0
    fi

    info "Cross-compiling YOLO runner..."

    docker run --rm \
        -v "${LLVM_INSTALL}:/llvm-install:ro" \
        -v "${ORT_SOURCE}:/onnxruntime:ro" \
        -v "${YOLO_RUNNER}:/runner:ro" \
        -v "${OUTPUT_DIR}/sysroot:/sysroot:ro" \
        -v "${ort_install}:/onnxruntime-install:ro" \
        -v "$(dirname "${runner_out}"):/out" \
        -e LLVM_INSTALL=/llvm-install \
        -e SYSROOT=/sysroot \
        "${DOCKER_IMAGE}" \
        bash -c "
            /llvm-install/bin/clang++-13 \
                --target=riscv64-unknown-linux-gnu \
                --sysroot=/sysroot \
                -std=c++17 -O2 -g \
                -I/onnxruntime-install/include \
                -I/onnxruntime-install/include/onnxruntime/core/session \
                -I/runner \
                /runner/yolo_runner.cpp \
                -o /out/yolo_inference \
                -L/onnxruntime-install/lib \
                -lonnxruntime \
                -Wl,-rpath,'\$ORIGIN/../onnxruntime/lib'
        "

    info "YOLO runner built: ${runner_out}"
}
```
